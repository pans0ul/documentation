47  /** \brief Intensity field accessor provides access to the intensity filed of a PoinT
48  * implementation for specific types should be done in \file pcl/common/impl/intensity.hpp
49  */
50  template<typename PointT>
51  struct IntensityFieldAccessor
52  {
53  /** \brief get intensity field
54  * \param[in] p point
55  * \return p.intensity
56  */
57  inline float
58  operator () (const PointT &p) const
59  {
60  return p.intensity;
61  }
62  /** \brief gets the intensity value of a point
63  * \param p point for which intensity to be get
64  * \param[in] intensity value of the intensity field
65  */
66  inline void
67  get (const PointT &p, float &intensity) const
68  {
69  intensity = p.intensity;
70  }
71  /** \brief sets the intensity value of a point
72  * \param p point for which intensity to be set
73  * \param[in] intensity value of the intensity field
74  */
75  inline void
76  set (PointT &p, float intensity) const
77  {
78  p.intensity = intensity;
79  }
80  /** \brief subtract value from intensity field
81  * \param p point for which to modify intensity
82  * \param[in] value value to be subtracted from point intensity
83  */
84  inline void
85  demean (PointT& p, float value) const
86  {
87  p.intensity -= value;
88  }
89  /** \brief add value to intensity field
90  * \param p point for which to modify intensity
91  * \param[in] value value to be added to point intensity
92  */
93  inline void
94  add (PointT& p, float value) const
95  {
96  p.intensity += value;
97  }
98  };
99  }
