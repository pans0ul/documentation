52  /* CopyPointHelper and its specializations copy the contents of a source
53  * point to a target point. There are three cases:
54  *
55  * - Points have the same type.
56  * In this case a single `memcpy` is used.
57  *
58  * - Points have different types and one of the following is true:
59  * * both have RGB fields;
60  * * both have RGBA fields;
61  * * one or both have no RGB/RGBA fields.
62  * In this case we find the list of common fields and copy their
63  * contents one by one with `NdConcatenateFunctor`.
64  *
65  * - Points have different types and one of these types has RGB field, and
66  * the other has RGBA field.
67  * In this case we also find the list of common fields and copy their
68  * contents. In order to account for the fact that RGB and RGBA do not
69  * match we have an additional `memcpy` to copy the contents of one into
70  * another.
71  *
72  * An appropriate version of CopyPointHelper is instantiated during
73  * compilation time automatically, so there is absolutely no run-time
74  * overhead. */
75 
76  template <typename PointInT, typename PointOutT, typename Enable = void>
77  struct CopyPointHelper { };

80  struct CopyPointHelper<PointInT, PointOutT, typename boost::enable_if<boost::is_same<PointInT, PointOutT> >::type>
81  {
82  void operator () (const PointInT& point_in, PointOutT& point_out) const
83  {
84  memcpy (&point_out, &point_in, sizeof (PointInT));
85  }
