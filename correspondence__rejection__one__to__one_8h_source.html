49  /** \brief CorrespondenceRejectorOneToOne implements a correspondence
50  * rejection method based on eliminating duplicate match indices in
51  * the correspondences. Correspondences with the same match index are
52  * removed and only the one with smallest distance between query and
53  * match are kept. That is, considering match->query 1-m correspondences
54  * are removed leaving only 1-1 correspondences.
55  * \author Dirk Holz
56  * \ingroup registration
57  */
58  class PCL_EXPORTS CorrespondenceRejectorOneToOne: public CorrespondenceRejector
59  {
60  using CorrespondenceRejector::input_correspondences_;
61  using CorrespondenceRejector::rejection_name_;
62  using CorrespondenceRejector::getClassName;
63 
64  public:
65  typedef boost::shared_ptr<CorrespondenceRejectorOneToOne> Ptr;
66  typedef boost::shared_ptr<const CorrespondenceRejectorOneToOne> ConstPtr;
67 
68  /** \brief Empty constructor. */
69  CorrespondenceRejectorOneToOne ()
70  {
71  rejection_name_ = "CorrespondenceRejectorOneToOne";
72  }
73 
74  /** \brief Get a list of valid correspondences after rejection from the original set of correspondences.
75  * \param[in] original_correspondences the set of initial correspondences given
76  * \param[out] remaining_correspondences the resultant filtered set of remaining correspondences
77  */
78  void
79  getRemainingCorrespondences (const pcl::Correspondences& original_correspondences,
80  pcl::Correspondences& remaining_correspondences);
81 
82  protected:
83  /** \brief Apply the rejection algorithm.
84  * \param[out] correspondences the set of resultant correspondences.
85  */
86  inline void
87  applyRejection (pcl::Correspondences &correspondences)
88  {
89  getRemainingCorrespondences (*input_correspondences_, correspondences);
90  }
91  };
