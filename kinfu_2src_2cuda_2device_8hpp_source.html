85  enum ScanKind { exclusive, inclusive };
86 
87  template<ScanKind Kind, class T>
88  __device__ __forceinline__ T
89  scan_warp ( volatile T *ptr, const unsigned int idx = threadIdx.x )
90  {
91  const unsigned int lane = idx & 31; // index of thread in warp (0..31)
92 
93  if (lane >= 1) ptr[idx] = ptr[idx - 1] + ptr[idx];
94  if (lane >= 2) ptr[idx] = ptr[idx - 2] + ptr[idx];
95  if (lane >= 4) ptr[idx] = ptr[idx - 4] + ptr[idx];
96  if (lane >= 8) ptr[idx] = ptr[idx - 8] + ptr[idx];
97  if (lane >= 16) ptr[idx] = ptr[idx - 16] + ptr[idx];
98 
99  if (Kind == inclusive)
100  return ptr[idx];
101  else
102  return (lane > 0) ? ptr[idx - 1] : 0;
103  }
104  }
