47 inline void
48 pcl::registration::getCorDistMeanStd (const pcl::Correspondences &correspondences, double &mean, double &stddev)
49 {
50  if (correspondences.empty ())
51  return;
52 
53  double sum = 0, sq_sum = 0;
54 
55  for (size_t i = 0; i < correspondences.size (); ++i)
56  {
57  sum += correspondences[i].distance;
58  sq_sum += correspondences[i].distance * correspondences[i].distance;
59  }
60  mean = sum / static_cast<double> (correspondences.size ());
61  double variance = (sq_sum - sum * sum / static_cast<double> (correspondences.size ())) / static_cast<double> (correspondences.size () - 1);
62  stddev = sqrt (variance);
63 }
