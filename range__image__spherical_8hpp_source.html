44 RangeImageSpherical::calculate3DPoint (float image_x, float image_y, float range, Eigen::Vector3f& point) const
45 {
46  float angle_x, angle_y;
47  getAnglesFromImagePoint (image_x, image_y, angle_x, angle_y);
48 
49  float cosY = cosf (angle_y);
50  point = Eigen::Vector3f (range * sinf (angle_x) * cosY, range * sinf (angle_y), range * cosf (angle_x)*cosY);
51  point = to_world_system_ * point;
52 }

55 inline void
56 RangeImageSpherical::getImagePoint (const Eigen::Vector3f& point, float& image_x, float& image_y, float& range) const
57 {
58  Eigen::Vector3f transformedPoint = to_range_image_system_ * point;
59  range = transformedPoint.norm ();
60  float angle_x = atan2LookUp (transformedPoint[0], transformedPoint[2]),
61  angle_y = asinLookUp (transformedPoint[1]/range);
62  getImagePointFromAngles (angle_x, angle_y, image_x, image_y);
63 }

66 RangeImageSpherical::getAnglesFromImagePoint (float image_x, float image_y, float& angle_x, float& angle_y) const
67 {
68  angle_y = (image_y+static_cast<float> (image_offset_y_))*angular_resolution_y_ - 0.5f*static_cast<float> (M_PI);
69  angle_x = ((image_x+ static_cast<float> (image_offset_x_))*angular_resolution_x_ - static_cast<float> (M_PI));
70 }

73 RangeImageSpherical::getImagePointFromAngles (float angle_x, float angle_y, float& image_x, float& image_y) const
74 {
75  image_x = (angle_x + static_cast<float> (M_PI))*angular_resolution_x_reciprocal_ - static_cast<float> (image_offset_x_);
76  image_y = (angle_y + 0.5f*static_cast<float> (M_PI))*angular_resolution_y_reciprocal_ - static_cast<float> (image_offset_y_);
77 }
